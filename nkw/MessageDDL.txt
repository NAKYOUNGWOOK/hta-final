create table message(
	no int auto_increment primary key,
	room int not null,
	sender int not null,
	receiver int not null,
	content varchar not null,
	writedate datetime not null,
	readcheck int not null,
	constraint m_sender foreign key(sender) references employee(empno),
	constraint m_receiver foreign key(receiver) references employee(empno),
);
